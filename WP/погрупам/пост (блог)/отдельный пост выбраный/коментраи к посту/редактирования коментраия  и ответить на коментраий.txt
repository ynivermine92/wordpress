// Обработчик Ajax для обновления комментариев
function update_comment_content() {
    // Проверка на наличие прав для редактирования комментариев
    if (!is_user_logged_in()) {
        wp_send_json_error(array('message' => 'Необходимо авторизоваться.'));
    }

    $comment_id = isset($_POST['comment_id']) ? intval($_POST['comment_id']) : 0;
    $new_content = isset($_POST['content']) ? sanitize_textarea_field($_POST['content']) : '';

    if ($comment_id && $new_content) {
        // Проверка прав пользователя на редактирование комментария
        $comment = get_comment($comment_id);
        if ($comment && (get_current_user_id() === $comment->user_id || current_user_can('moderate_comments'))) {
            // Обновляем комментарий
            wp_update_comment(array(
                'comment_ID' => $comment_id,
                'comment_content' => $new_content
            ));

            // Отправляем успешный ответ
            wp_send_json_success();
        }
    }

    // Если ошибка, отправляем сообщение об ошибке
    wp_send_json_error(array('message' => 'Ошибка обновления комментария.'));
}

// Регистрируем обработчик для Ajax
add_action('wp_ajax_update_comment_content', 'update_comment_content'); // Для авторизованных пользователей
add_action('wp_ajax_nopriv_update_comment_content', 'update_comment_content'); // Для неавторизованных пользователей

	/* добавить редактировать коментраий  на странице и запрос  серверу  */
	const comments = () => {
		document.addEventListener('DOMContentLoaded', function () {
			// Показываем форму редактирования
			document.querySelectorAll('.comment-edit-link').forEach(function (editLink) {
				editLink.addEventListener('click', function (e) {
					e.preventDefault();
					let commentId = this.getAttribute('data-comment-id');
					let editForm = document.getElementById('edit-comment-' + commentId);
					if (editForm) {
						editForm.style.display = editForm.style.display === 'none' ? 'block' : 'none';
					}
				});
			});
	
			// Отправляем обновленный комментарий на сервер
			document.querySelectorAll('.save-comment').forEach(function (saveButton) {
				saveButton.addEventListener('click', function () {
					let commentId = this.getAttribute('data-comment-id');
					let newContent = document.querySelector('#edit-comment-' + commentId + ' textarea').value;
	
					// Проверяем, что контент не пустой
					if (newContent.trim() === '') {
						alert('Комментарий не может быть пустым');
						return;
					}
	
					// Создаем новый запрос AJAX с использованием fetch
					fetch(commentEditData.ajaxurl, {
						method: 'POST',
						headers: {
							'Content-Type': 'application/x-www-form-urlencoded'
						},
						body: 'action=update_comment_content&comment_id=' + commentId + '&content=' + encodeURIComponent(newContent)
					})
					.then(response => response.json())
					.then(data => {
						if (data.success) {
							// Обновляем текст комментария на странице
							let commentContent = document.querySelector('#comment-' + commentId + ' .comment-content');
							if (commentContent) {
								commentContent.textContent = newContent;
							}
							let editForm = document.getElementById('edit-comment-' + commentId);
							if (editForm) {
								editForm.style.display = 'none'; // Скрываем форму редактирования
							}
						} else {
							alert('Ошибка обновления комментария');
						}
					})
					.catch(error => {
						console.error('Ошибка:', error);
						alert('Ошибка отправки комментария');
					});
				});
			});
		});
	};
	
	comments();

                // Начинаем вывод комментария с кастомным классом

                echo '<div class="vcard bio">' . $avatar . '</div>'; // Добавляем аватар
                echo '<div class="comment-body">';

                echo '<div class="meta">' . $date . ' at ' . $time . '</div>'; // Дата и время
