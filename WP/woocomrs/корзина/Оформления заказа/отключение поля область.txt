ОПХМСДХРЕКЭМН НРЙКЧВХРЭ НАЪГЮРЕКЭМНЯРЭ ОНКЪ «нАКЮЯРЭ»  ( ВРН АШ ОНЙСОЙЮ ОПНЬКЮ ДЮКЭЬЕ )


УСЙ НРЙКЧВЮЕР ОПХМСДЕКЭМН ОНКЕ НАКЮЯРЭ ( ЕЦН Х РЮЙ МЕР ОН ДЕТНКРС)

add_filter( 'woocommerce_default_address_fields', 'custom_optional_state_field' );
function custom_optional_state_field( $address_fields ) {
    if ( isset( $address_fields['state'] ) ) {
        $address_fields['state']['required'] = false;
    }
    return $address_fields;
}
